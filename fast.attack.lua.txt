function DanhNhanhBuCu36.HeadExtract(list)
    local result = {}
    local c = #list

--67

    for i = 1, c do
        local v = list[i]
        local h = v:FindFirstChild("Head") or v.PrimaryPart
        if h then
            result[#result+1] = {v, h}
        end
    end
    return result
end
